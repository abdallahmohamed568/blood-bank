📋 كود Apps Script المطلوب في جدولك

    

      تأكد أن هذا الكود موجود في Extensions → Apps Script
      وأنك غيّرت YOUR_SHEET_ID_HERE بـ 1by4_xeHlL4qKTPIJ7Lfz9U80F5JK1pFR1iJr87a_Bu8
    

    
    
const SHEET_ID = '1by4_xeHlL4qKTPIJ7Lfz9U80F5JK1pFR1iJr87a_Bu8';

function doPost(e) {
  try {
    var ss   = SpreadsheetApp.openById(SHEET_ID);
    var body = JSON.parse(e.postData.contents);
    var tab  = ss.getSheetByName(body.sheet);
    if (!tab) {
      tab = ss.insertSheet(body.sheet);
      var hdrs = getHeaders(body.sheet);
      tab.appendRow(hdrs);
      tab.getRange(1,1,1,hdrs.length)
         .setBackground('#C0392B')
         .setFontColor('#FFFFFF')
         .setFontWeight('bold');
      tab.setFrozenRows(1);
    }
    tab.appendRow(body.data.map(function(v){
      return v===undefined||v===null?'':v;
    }));
    return out({status:'ok'});
  } catch(err) {
    return out({status:'error', msg:err.message});
  }
}

function doGet(e) {
  try {
    var ss   = SpreadsheetApp.openById(SHEET_ID);
    var tab  = ss.getSheetByName(e.parameter.sheet||'المستفيدون');
    if (!tab) return out({status:'error', msg:'Sheet not found'});
    var data = tab.getDataRange().getValues();
    return out({status:'ok', data:data});
  } catch(err) {
    return out({status:'error', msg:err.message});
  }
}

function out(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function getHeaders(sheet) {
  var h = {
    'المستفيدون':  ['التاريخ','المعرف','الاسم','فصيلة الدم','الهاتف','البريد','العنوان','صورة الإثبات','الحالة'],
    'المتبرعون':   ['التاريخ','المعرف','الاسم','فصيلة الدم','الهاتف','البريد','العنوان','مدخن','أمراض','تفاصيل','آخر تبرع','الحالة'],
    'المستشفيات':  ['التاريخ','المعرف','اسم المستشفى','العنوان','البريد','الهاتف','رقم الترخيص','النوع','ملاحظات','الحالة'],
    'الدردشة':     ['التاريخ','معرف المرسل','اسم المرسل','معرف المستقبل','اسم المستقبل','الرسالة','النوع','الحالة']
  };
  return h[sheet] || ['التاريخ','البيانات'];
}
